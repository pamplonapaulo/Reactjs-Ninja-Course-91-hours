'use strict'
 
import mainReduce from './prof-main-reduce'
 
const reduceRight = (arr, func, acc) => {
 
  if (acc === undefined) {
    acc = arr[arr.length - 1]
    arr = arr.slice(0, arr.length - 1)
  }
 
  for (let i = arr.length - 1; i >= 0; i--)
    acc = func(acc, arr[i], i, arr)    
 
  return acc
}
 
export default reduceRight
